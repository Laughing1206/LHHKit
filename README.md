LHHKit目录
```
LHHKit
  |___LHHKitHeader.h //header文件
  |___MacroHeader.h //常用宏
  |___FrameWork  //FrameWork
  	|___AFNetworking  //AFNetworking
 	|___AppService  //AppService
		|___AppConstant  //常量
		|___AppConfig  //初始化配置
		|___AppDelegateProxy  //伪多继承
		|___AppService  //继承AppService，子类使用AppDelegate的代理方法
		|___AppTheme  //tools常用方法
	|___AppSplash  //广告页
	|___Aspects  //统一处理逻辑，不改变源码的情况下，插入代码
	|___Authorization  //登录相关
	|___AutoCoding  //数模转换
	|___BlocksKit  //BlocksKit
	|___Cache  //缓存
	|___CheckManager  //版本更新
	|___CheckNumBtn  //验证码倒计时
	|___DeepLink  //路由
	|___ErrorHandler  //错误处理
	|___Extension  //UIKit+Foundation的相关类目
	|___FullscreenPopGesture  //全屏滑动
	|___LHHSignal  //signal，给UIview，UIControl等添加响应
	|___LHHSystemInfo  //关于系统的几个方法
	|___MailLikeTransitionAnimator  //仿支付宝选择属性的转场动画
	|___Masonry  //代码布局
	|___MBProgressHUD  //HUD
	|___MJRefresh  //上拉加载，下拉刷新
	|___Model  // 模型
	|___NetWork  //网络请求
	|___NibStyle  //控件添加属性
	|___NumberButton  //商品加减按钮
	|___PageTopView  //滑动菜单+菜单
	|___Payment  //支付宝+微信+银联支付
	|___RichScan  //扫码
	|___RouterParser  //路由
	|___ServiceNetworkLogger  //debuger
	|___ShareService  //微信+QQ+微博分享
	|___SSKeychain  //SSKeychain
	|___TableViewRowAction  //tableView添加左滑多个按钮（原生），适配iOS6，7
	|___Transformer  //加密+数模转换
	|___TYAlertController  //提示弹框
	|___UICollectonViewLayout  //默认居左+居左+居右+瀑布流
	|___UIPlaceHolderTextView  //TextView添加PlaceHolder
	|___Validator  //正则
```
